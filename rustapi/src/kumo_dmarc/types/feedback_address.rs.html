9impl FromStr for FeedbackAddress {

11
12    fn from_str(s: &str) -> Result<Self, Self::Err> {
13        let s = s.trim();
14        if s.is_empty() {
15            return Err("empty feedback address {s:?}".to_owned());
16        }
17
18        let Some((uri, size)) = s.trim().rsplit_once('!') else {
19            return Ok(Self {
20                uri: s.to_owned(),
21                size: None,
22            });
23        };
24
25        let size = size.trim();
26        if size.is_empty() {
27            return Err(format!("empty size in {s:?}"));
28        }
29
30        let mut power = 0;
31        match size.chars().next_back() {
32            Some('k') => power = 10,
33            Some('m') => power = 20,
34            Some('g') => power = 30,
35            Some('t') => power = 40,
36            _ => {}
37        }
38
39        let size = match power {
40            0 => size,
41            _ => &size[..size.len() - 1],
42        };
43
44        let size = u64::from_str(size).map_err(|_| format!("invalid size in {s:?}"))? << power;
45        Ok(Self {
46            uri: uri.to_owned(),
47            size: Some(size),
48        })
49    }
50}
