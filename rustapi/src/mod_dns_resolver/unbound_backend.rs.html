7/// The fields from kumomta's `ResolverOptions` that map to an unbound
8/// configuration call. Any other field set on the public options struct
9/// is rejected at configure time via serde's `deny_unknown_fields` when
10/// re-deserializing the JSON form of the public options into this one.
11#[derive(Debug, Default, serde::Deserialize)]
12#[serde(deny_unknown_fields)]
13struct UnboundResolverOptions {
14    #[serde(default)]
15    validate: Option<bool>,
16    #[serde(default)]
17    trust_anchor_file: Option<TrustAnchorFile>,
18    #[serde(default)]
19    use_hosts_file: Option<UseHostsFile>,
20}
21
22impl UnboundResolverOptions {
23    fn from_public(opts: &crate::config::ResolverOptions) -> anyhow::Result<Self> {
24        let value = serde_json::to_value(opts)
25            .expect("ResolverOptions Serialize impl should be infallible");
26        serde_json::from_value(value)
27            .map_err(|e| anyhow::anyhow!("option not supported by the unbound backend: {e}"))
28    }
29}

31pub fn build_unbound_resolver(config: &DnsResolverConfig) -> anyhow::Result<UnboundResolver> {
32    let opts = UnboundResolverOptions::from_public(&config.options)?;
33
34    let context = libunbound::Context::new()?;
35
36    // libunbound chooses UDP/TCP internally per query and has a single
37    // forwarder list, so multiple entries for the same upstream IP would
38    // register that upstream more than once.
39    let mut seen = BTreeSet::new();
40    for ns in &config.name_servers {
41        let addr = ns_socket_addr(ns)?;
42        if seen.insert(addr) {
43            context.set_forward(Some(addr)).context("set_forward")?;
44        }
45    }
46
47    match &opts.trust_anchor_file {
48        // No anchor file: validate against the bundled root trust anchors.
49        None => {
50            if opts.validate.unwrap_or(false) {
51                context
52                    .add_builtin_trust_anchors()
53                    .context("add_builtin_trust_anchors")?;
54            }
55        }
56        // Static file: loaded in addition to the bundled anchors (when
57        // validation is enabled), and never updated.
58        Some(TrustAnchorFile::Static(path)) => {
59            if opts.validate.unwrap_or(false) {
60                context
61                    .add_builtin_trust_anchors()
62                    .context("add_builtin_trust_anchors")?;
63            }
64            context
65                .load_trust_anchor_file(path)
66                .context("load_trust_anchor_file")?;
67        }
68        // Managed file: RFC 5011 auto-maintained. The file is the sole anchor
69        // source (no bundled anchors); seed it with the bundled anchors if it
70        // is missing or empty so validation works immediately and RFC 5011
71        // keeps it current thereafter.
72        Some(TrustAnchorFile::Managed { managed }) => {
73            seed_managed_anchor_file(managed)?;
74            context
75                .load_trust_anchor_file_with_auto_update(managed)
76                .context("load_trust_anchor_file_with_auto_update")?;
77        }
78    }
79
80    if !matches!(opts.use_hosts_file, Some(UseHostsFile::Never)) {
81        context.load_hosts(None).context("load_hosts")?;
82    }
83
84    let context = context
85        .into_async()
86        .context("make async resolver context")?;
87    Ok(UnboundResolver::from(context))
88}

90/// Seed an RFC 5011 managed trust anchor file with the bundled root anchors if
91/// it does not yet exist (or is empty). This gives unbound a valid starting
92/// anchor so DNSSEC validation works on first use; RFC 5011 then maintains the
93/// file across future root KSK rollovers.
94fn seed_managed_anchor_file(path: &str) -> anyhow::Result<()> {
95    let needs_seed = match std::fs::metadata(path) {
96        Ok(meta) => meta.len() == 0,
97        Err(_) => true,
98    };
99    if needs_seed {
100        std::fs::write(path, libunbound::ROOT_TRUST_ANCHORS.join("\n"))
101            .with_context(|| format!("seeding managed trust anchor file {path:?}"))?;
102    }
103    Ok(())
104}

106fn ns_socket_addr(ns: &NameServer) -> anyhow::Result<SocketAddr> {
107    let s = match ns {
108        NameServer::Ip(s) => s.as_str(),
109        NameServer::Detailed { socket_addr, .. } => socket_addr.as_str(),
110    };
111    s.parse::<SocketAddr>()
112        .with_context(|| format!("name server: '{s}'"))
113}

115#[cfg(test)]
116mod tests {
117    use super::*;
118    use crate::config::ResolverOptions;
119
120    #[test]
121    fn supported_options_round_trip() {
122        let opts = ResolverOptions {
123            validate: Some(true),
124            use_hosts_file: Some(UseHostsFile::Auto),
125            ..ResolverOptions::default()
126        };
127        let unbound_opts = UnboundResolverOptions::from_public(&opts).unwrap();
128        assert_eq!(unbound_opts.validate, Some(true));
129        assert!(matches!(
130            unbound_opts.use_hosts_file,
131            Some(UseHostsFile::Auto)
132        ));
133    }
134
135    #[test]
136    fn unsupported_option_is_rejected() {
137        let opts = ResolverOptions {
138            ndots: Some(2),
139            ..ResolverOptions::default()
140        };
141        let err = UnboundResolverOptions::from_public(&opts)
142            .unwrap_err()
143            .to_string();
144        assert_eq!(
145            err,
146            "option not supported by the unbound backend: unknown field `ndots`, \
147             expected one of `validate`, `trust_anchor_file`, `use_hosts_file`",
148        );
149    }
