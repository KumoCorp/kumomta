8/// The fields from kumomta's `ResolverOptions` that map to an unbound
9/// configuration call. Any other field set on the public options struct
10/// is rejected at configure time via serde's `deny_unknown_fields` when
11/// re-deserializing the JSON form of the public options into this one.
12#[derive(Debug, Default, serde::Deserialize)]
13#[serde(deny_unknown_fields)]
14struct UnboundResolverOptions {
15    #[serde(default)]
16    validate: Option<bool>,
17    #[serde(default)]
18    trust_anchor_file: Option<PathBuf>,
19    #[serde(default)]
20    use_hosts_file: Option<UseHostsFile>,
21}
22
23impl UnboundResolverOptions {
24    fn from_public(opts: &crate::config::ResolverOptions) -> anyhow::Result<Self> {
25        let value = serde_json::to_value(opts)
26            .expect("ResolverOptions Serialize impl should be infallible");
27        serde_json::from_value(value)
28            .map_err(|e| anyhow::anyhow!("option not supported by the unbound backend: {e}"))
29    }
30}

32pub fn build_unbound_resolver(config: &DnsResolverConfig) -> anyhow::Result<UnboundResolver> {
33    let opts = UnboundResolverOptions::from_public(&config.options)?;
34
35    let context = libunbound::Context::new()?;
36
37    // libunbound chooses UDP/TCP internally per query and has a single
38    // forwarder list, so multiple entries for the same upstream IP would
39    // register that upstream more than once.
40    let mut seen = BTreeSet::new();
41    for ns in &config.name_servers {
42        let addr = ns_socket_addr(ns)?;
43        if seen.insert(addr) {
44            context.set_forward(Some(addr)).context("set_forward")?;
45        }
46    }
47
48    if opts.validate.unwrap_or(false) {
49        context
50            .add_builtin_trust_anchors()
51            .context("add_builtin_trust_anchors")?;
52    }
53
54    if let Some(path) = &opts.trust_anchor_file {
55        let path_str = path
56            .to_str()
57            .ok_or_else(|| anyhow::anyhow!("trust_anchor_file path is not valid UTF-8"))?;
58        context
59            .load_trust_anchor_file(path_str)
60            .context("load_trust_anchor_file")?;
61    }
62
63    if !matches!(opts.use_hosts_file, Some(UseHostsFile::Never)) {
64        context.load_hosts(None).context("load_hosts")?;
65    }
66
67    let context = context
68        .into_async()
69        .context("make async resolver context")?;
70    Ok(UnboundResolver::from(context))
71}

73fn ns_socket_addr(ns: &NameServer) -> anyhow::Result<SocketAddr> {
74    let s = match ns {
75        NameServer::Ip(s) => s.as_str(),
76        NameServer::Detailed { socket_addr, .. } => socket_addr.as_str(),
77    };
78    s.parse::<SocketAddr>()
79        .with_context(|| format!("name server: '{s}'"))
80}

82#[cfg(test)]
83mod tests {
84    use super::*;
85    use crate::config::ResolverOptions;
86
87    #[test]
88    fn supported_options_round_trip() {
89        let opts = ResolverOptions {
90            validate: Some(true),
91            use_hosts_file: Some(UseHostsFile::Auto),
92            ..ResolverOptions::default()
93        };
94        let unbound_opts = UnboundResolverOptions::from_public(&opts).unwrap();
95        assert_eq!(unbound_opts.validate, Some(true));
96        assert!(matches!(
97            unbound_opts.use_hosts_file,
98            Some(UseHostsFile::Auto)
99        ));
100    }
101
102    #[test]
103    fn unsupported_option_is_rejected() {
104        let opts = ResolverOptions {
105            ndots: Some(2),
106            ..ResolverOptions::default()
107        };
108        let err = UnboundResolverOptions::from_public(&opts)
109            .unwrap_err()
110            .to_string();
111        assert_eq!(
112            err,
113            "option not supported by the unbound backend: unknown field `ndots`, \
114             expected one of `validate`, `trust_anchor_file`, `use_hosts_file`",
115        );
116    }
